```typescript
// Types
export interface ServiceGroup {
  id: number
  name: string
  description: string | null
  color: string
  sort_order: number
  member_count: number
  created_at: string
  updated_at: string
}

export interface ServiceGroupCreate {
  name: string
  description?: string | null
  color?: string
  sort_order?: number
}

export interface ServiceGroupUpdate {
  name?: string
  description?: string | null
  color?: string
  sort_order?: number
}

export interface SenaiteAnalyst {
  username: string
  fullname: string
}

// API functions
export async function getServiceGroups(): Promise<ServiceGroup[]> {
  const resp = await fetch(`${BASE_URL}/service-groups`, fetchOpts())
  if (!resp.ok) throw new Error('Failed to load service groups')
  return resp.json()
}

export async function createServiceGroup(data: ServiceGroupCreate): Promise<ServiceGroup> {
  const resp = await fetch(`${BASE_URL}/service-groups`, {
    ...fetchOpts(),
    method: 'POST',
    headers: { ...fetchOpts().headers, 'Content-Type': 'application/json' },
    body: JSON.stringify(data),
  })
  if (!resp.ok) throw new Error('Failed to create service group')
  return resp.json()
}

export async function updateServiceGroup(id: number, data: ServiceGroupUpdate): Promise<ServiceGroup> {
  const resp = await fetch(`${BASE_URL}/service-groups/${id}`, {
    ...fetchOpts(),
    method: 'PUT',
    headers: { ...fetchOpts().headers, 'Content-Type': 'application/json' },
    body: JSON.stringify(data),
  })
  if (!resp.ok) throw new Error('Failed to update service group')
  return resp.json()
}

export async function deleteServiceGroup(id: number): Promise<void> {
  const resp = await fetch(`${BASE_URL}/service-groups/${id}`, {
    ...fetchOpts(),
    method: 'DELETE',
  })
  if (!resp.ok) throw new Error('Failed to delete service group')
}

export async function setServiceGroupMembers(groupId: number, analysisServiceIds: number[]): Promise<{ count: number }> {
  const resp = await fetch(`${BASE_URL}/service-groups/${groupId}/members`, {
    ...fetchOpts(),
    method: 'PUT',
    headers: { ...fetchOpts().headers, 'Content-Type': 'application/json' },
    body: JSON.stringify({ analysis_service_ids: analysisServiceIds }),
  })
  if (!resp.ok) throw new Error('Failed to update service group members')
  return resp.json()
}

export async function getSenaiteAnalysts(): Promise<SenaiteAnalyst[]> {
  const resp = await fetch(`${BASE_URL}/senaite/analysts`, fetchOpts())
  if (!resp.ok) throw new Error('Failed to load analysts')
  return resp.json()
}

export async function setAnalysisAnalyst(uid: string, analystValue: string): Promise<{ success: boolean; analyst_stored: string | null }> {
  const resp = await fetch(`${BASE_URL}/senaite/analyses/${uid}/analyst`, {
    ...fetchOpts(),
    method: 'POST',
    headers: { ...fetchOpts().headers, 'Content-Type': 'application/json' },
    body: JSON.stringify({ analyst_value: analystValue }),
  })
  if (!resp.ok) throw new Error('Failed to assign analyst')
  return resp.json()
}
```
